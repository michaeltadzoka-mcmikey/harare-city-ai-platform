function showBulkRestoreModal(docIds) { let modal = document.getElementById('bulkRestoreModal'); if (!modal) { // Create modal if not present const modalHTML = `

♻️ Bulk Restore Documents

Restoring archived documents.

`; document.body.insertAdjacentHTML('beforeend', modalHTML); modal = document.getElementById('bulkRestoreModal'); // reassign after creation } // Update count document.getElementById('bulkRestoreCount').textContent = docIds.length; // Set default date document.getElementById('bulkRestoreValidFrom').value = new Date().toISOString().split('T')[0]; modal.style.display = 'block'; // Remove previous listener and attach new one const executeBtn = document.getElementById('executeBulkRestore'); const newBtn = executeBtn.cloneNode(true); executeBtn.parentNode.replaceChild(newBtn, executeBtn); newBtn.addEventListener('click', () => { const data = { action: 'restore', document_ids: docIds, valid_from: document.getElementById('bulkRestoreValidFrom').value, valid_to: document.getElementById('bulkRestoreValidTo').value || null, mode: document.querySelector('input[name="bulkRestoreMode"]:checked').value }; fetch('/documents/api/bulk', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(data) }) .then(res => res.json()) .then(data => { if (data.success) { alert(`Bulk restore completed on ${data.count} documents`); modal.style.display = 'none'; loadList(currentPage); selectedDocs.clear(); document.getElementById('selectAllCheckbox').checked = false; refreshTree(); fetch('/documents/api/filter-options') .then(res => res.json()) .then(filterData => populateFilters(filterData)); } else { alert('Bulk restore failed'); } }); }); }
