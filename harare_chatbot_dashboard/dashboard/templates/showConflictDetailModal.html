function showConflictDetailModal(service) { fetch(`/documents/api/conflict-details?service=${service}`) .then(res => res.json()) .then(data => { let modal = document.getElementById('conflictDetailModal'); if (!modal) { const modalHTML = `

⚔️ Conflict Details

Document 1 Title Document 2 Title Actions
`; document.body.insertAdjacentHTML('beforeend', modalHTML); modal = document.getElementById('conflictDetailModal'); } const tbody = document.getElementById('conflictDetailBody'); tbody.innerHTML = ''; if (data.conflicts && data.conflicts.length > 0) { data.conflicts.forEach(c => { const row = tbody.insertRow(); row.innerHTML = ` ${c.doc1_id} ${c.doc1_title} ${c.doc2_id} ${c.doc2_title} `; }); } else { tbody.innerHTML = 'No conflicts found for this service.'; } modal.style.display = 'block'; }); }
